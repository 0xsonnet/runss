import { useState } from 'react'

```jsx
function WindTerminal() {
  const [output, setOutput] = useState([
    'simulator@ethereal-node:~/$ wind init',
    '[+] summoning genesis block...',
    '[🌀] collecting entropy from ambient noise... ok',
    '[🌪️] engraving sigil into chainspace... done',
    '[🌬️] WIND contract deployed at: ',
    '',
    'ascii sigil projection:',
    '',
    '      ~~~ WIND TOKEN SIGIL ~~~',
    '       )\\   )\\     /)       /(',
    '      /(_)_( )\\___/( )__/\\_ )_\\___',
    '     (    /  (     )  (    )     /',
    '      \\__/    \\___/    \\__/_____/',
  ])

  const handleCommand = (cmd) => {
    if (cmd === 'wind mint 333') {
      setOutput(prev => [
        ...prev,
        'simulator@ethereal-node:~/$ wind mint 333',
        '[+] invoking breath function...',
        '[🔮] oracle gas bypass active',
        '[✨] minted 333 WIND tokens to address: ',
      ])
    } else if (cmd === 'wind whisper') {
      setOutput(prev => [
        ...prev,
        'simulator@ethereal-node:~/$ wind whisper',
        '[👁️] decoding oracle:',
        '"do not store WIND, become it."',
      ])
    } else {
      setOutput(prev => [
        ...prev,
        `simulator@ethereal-node:~/$ ${cmd}`,
        'unknown command... the æther does not respond.',
      ])
    }
  }

  return (
    <div className="bg-black text-green-400 font-mono p-4 rounded-xl shadow-md mb-6">
      {output.map((line, idx) => (
        <div key={idx}>{line}</div>
      ))}
      <input
        className="bg-transparent border-none outline-none mt-2 text-green-200 w-full"
        placeholder="type wind command..."
        onKeyDown={(e) => {
          if (e.key === 'Enter') {
            handleCommand(e.target.value)
            e.target.value = ''
          }
        }}
      />
    </div>
  )
}

<WindTerminal />
